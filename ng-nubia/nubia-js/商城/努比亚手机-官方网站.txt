    var NUBIA_STORE_COMMON={

            _nb_mc_item.on({

                    var $this=$(this);
                    var left_cl=240;
                    var top_cl;

                            $this.find('.nb-mlside-menu').css("width","720px");

                                $this.find('.nb-mlside-menu').css("width","960px");
                                if(cl_num.length>24){
                                    $this.find('.nb-mlside-menu').css("width","1200px");
                                }
                            }
                        }

                            $this.find('.nb-mlside-menu>a').eq(i).css({
                                "position":"absolute",
                                "left":left_cl+"px",
                                "top":top_cl*89+"px"
                            });
                        }
                    }


                },
                mouseleave : function(){
                    $(this).children('a').removeClass('current');
                    $(this).find('.nb-mlside-menu').css({display:'none'});

                    $price.toggle()
                });
            });
